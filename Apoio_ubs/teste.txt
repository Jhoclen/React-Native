import { agendaMock, getUniqueDates, getTimeSlotsForDate } from './mockData';

export default function DateTimeSelectorGeneric({ 
  servicoSelecionado, // Ex: "Vacina", "Consulta"
  onConfirm // Callback quando confirmar
}) {
  const [selectedDate, setSelectedDate] = useState(null);
  const [selectedTime, setSelectedTime] = useState(null);

  // üéØ FILTRA APENAS AS DATAS DESTE SERVI√áO
  const datasDisponiveis = useMemo(() => {
    return getUniqueDates(servicoSelecionado);
  }, [servicoSelecionado]);

  // üéØ FILTRA APENAS OS HOR√ÅRIOS DA DATA SELECIONADA
  const horariosDisponiveis = useMemo(() => {
    if (!selectedDate) return [];
    return getTimeSlotsForDate(servicoSelecionado, selectedDate);
  }, [servicoSelecionado, selectedDate]);

  const handleDateSelect = (data) => {
    setSelectedDate(data);
    setSelectedTime(null); // Reseta hor√°rio ao trocar de dia
  };

  const handleTimeSelect = (horario) => {
    if (horario.disponibilidade) {
      setSelectedTime(horario);
    }
  };

  const handleConfirm = () => {
    if (selectedDate && selectedTime) {
      onConfirm({
        servico: servicoSelecionado,
        data: selectedDate,
        hora: selectedTime.hora,
        agendamentoId: selectedTime.id
      });
    }
  };

  // Formata data para exibi√ß√£o (15/01)
  const formatDate = (dateString) => {
    const [year, month, day] = dateString.split('-');
    return `${day}/${month}`;
  };

  // Pega nome do dia da semana
  const getDayName = (dateString) => {
    const days = ['Dom', 'Seg', 'Ter', 'Qua', 'Qui', 'Sex', 'S√°b'];
    const date = new Date(dateString + 'T00:00:00');
    return days[date.getDay()];
  };

  return (
    <View style={styles.container}>
      {/* ===== T√çTULO DO SERVI√áO ===== */}
      <View style={styles.header}>
        <Text style={styles.serviceTitle}>{servicoSelecionado}</Text>

      </View>

      {/* ===== SE√á√ÉO 1: ESCOLHER DIA ===== */}
      <Text style={styles.sectionTitle}>üìÖ Escolha o Dia</Text>
      
      <ScrollView 
        horizontal 
        showsHorizontalScrollIndicator={false}
        style={styles.datesScroll}
      >
        {datasDisponiveis.map((data) => {
          const isSelected = selectedDate === data;
          
          return (
            <TouchableOpacity
              key={data}
              onPress={() => handleDateSelect(data)}
              style={[
                styles.dateCard,
                isSelected && styles.dateCardSelected
              ]}
            >
              <Text style={[styles.dayName, isSelected && styles.textSelected]}>
                {getDayName(data)}
              </Text>
              <Text style={[styles.dayNumber, isSelected && styles.textSelected]}>
                {data.split('-')[2]}
              </Text>
              <Text style={[styles.month, isSelected && styles.textSelected]}>
                {formatDate(data)}
              </Text>
            </TouchableOpacity>
          );
        })}
      </ScrollView>

      {/* ===== SE√á√ÉO 2: ESCOLHER HOR√ÅRIO (S√ì APARECE SE TEM DIA) ===== */}
      {selectedDate && horariosDisponiveis.length > 0 && (
        <View style={styles.timeSlotsSection}>
          <Text style={styles.sectionTitle}>üïê Escolha o Hor√°rio</Text>
          
          <View style={styles.timeSlotsGrid}>
            {horariosDisponiveis.map((horario) => {
              const isSelected = selectedTime?.id === horario.id;
              const isAvailable = horario.disponibilidade;
              
              return (
                <TouchableOpacity
                  key={horario.id}
                  onPress={() => handleTimeSelect(horario)}
                  disabled={!isAvailable}
                  style={[
                    styles.timeSlot,
                    isSelected && styles.timeSlotSelected,
                    !isAvailable && styles.timeSlotDisabled
                  ]}
                >
                  <Text style={[
                    styles.timeText,
                    isSelected && styles.timeTextSelected,
                    !isAvailable && styles.timeTextDisabled
                  ]}>
                    {horario.hora}
                  </Text>
                  {!isAvailable && (
                    <Text style={styles.ocupadoText}>Ocupado</Text>
                  )}
                </TouchableOpacity>
              );
            })}
          </View>
        </View>
      )}

      {/* ===== MENSAGEM SE N√ÉO TEM HOR√ÅRIOS ===== */}
      {selectedDate && horariosDisponiveis.length === 0 && (
        <View style={styles.emptyState}>
          <Text style={styles.emptyText}>
            Nenhum hor√°rio dispon√≠vel nesta data
          </Text>
        </View>
      )}

      {/* ===== BOT√ÉO CONFIRMAR (S√ì APARECE SE TEM DIA E HORA) ===== */}
      {selectedDate && selectedTime && (
        <TouchableOpacity 
          style={styles.confirmButton}
          onPress={handleConfirm}
        >
          <Text style={styles.confirmText}>
            Confirmar {servicoSelecionado}
          </Text>
          <Text style={styles.confirmDetails}>
            {formatDate(selectedDate)} √†s {selectedTime.hora}
          </Text>
        </TouchableOpacity>
      )}
    </View>
  );
}
